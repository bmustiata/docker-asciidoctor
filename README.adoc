= Asciidoctor Docker Container

Versions bundled:

Asciidoctor: 1.5.8
Asciidoctor PDF: 1.5.0.alpha.17.dev

[source,sh]
-----------------------------------------------------------------------------
apt-get update -y && \
apt-get install -y \
    asciidoctor-doc \
    ruby-asciidoctor \
    ruby-asciidoctor-plantuml \
    ruby-asciidoctor-pdf && \
gem install asciidoctor-diagram asciidoctor-plantuml
